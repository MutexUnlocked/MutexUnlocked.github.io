Mutex Unlocked

2019-08-17 :: Friedrich Doku 
// Mutex Locked
#include <iostream>
#include <vector>
#include <mutex>
#include <thread>
std::mutex data;
std::vector<int> myvec;

void run(){
    for(int i = 0; i < 10; i++){
        data.lock();
        myvec.push_back(i);
        data.unlock();
    }
}

int
main(){
    std::thread t1(run);
    std::thread t2(run);
    t1.join();
    t2.join();
    return 0;
}
